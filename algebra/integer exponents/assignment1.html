\[ \frac{n^{-2}m}{7m^{-4}n^{-3}} = \frac{\frac{m}{n^2}}{\frac{7}{m^4n^3}} = \frac{m}{n^2} . \frac{m^4n^3}{7} = \frac{m^5n}{7} \]

\[ \frac{5x^{-1}y^{-4}}{(3y^5)^{-2}x^9} = \frac{5x^{-1}y^{-4}}{(3^{-2}y^{-2.5})x^9} = \frac{\frac{5}{xy^4}}{\frac{x^9}{9.y^{10}}} = \frac{5}{xy^4} . \frac{9.y^{10}}{x^9} = \frac{45y^6}{x^{10}} \]

\[ \left(\frac{z^{-5}}{z^{-2}x^{-1}}\right)^6 = \left(\frac{\frac{1}{z^5}}{\frac{1}{z^2.x}}\right)^6 = \left(\frac{1}{z^5}.\frac{z^2.x}{1}\right)^6 = \left(\frac{x}{z^3}\right)^6 = \frac{x^6}{z^{18}} \]

\[ \left( \frac{24a^3b^{-8}}{6a^{-5}b} \right)^{-2} = \left( \frac{\frac{24a^3}{b^{8}}}{\frac{6b}{a^{5}}} \right)^{-2} = \left( \frac{24a^3}{b^{8}} . \frac{a^{5}}{6b} \right)^{-2} = \left( \frac{4a^{8}}{b^{9}} \right)^{-2} =\frac{4^{-2}a^{8.-2}}{b^{9.-2}} = \frac{\frac{1}{16.a^{16}}}{\frac{1}{b^{18}}} = \frac{b^{18}}{16.a^{16}} \]
