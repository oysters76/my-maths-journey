Problem Set: Rational Expressions - Problem Set 1

\[ \frac{x^2 - 2x - 8}{x^2 - 9x + 20} = \frac{(x - 4)(x + 2)}{(x - 5)(x-4)} = \frac{\cancel{(x - 4)}(x + 2)}{(x - 5)\cancel{(x-4)}} = \frac{x + 2}{x - 5} \]

\[ \frac{x^2 - 25}{5x - x^2} = \frac{(x-5)(x+5)}{-x(-5+x)} = \frac{(x-5)(x+5)}{-x(x-5)} = \frac{\cancel{(x-5)}(x+5)}{-x\cancel{(x-5)}} = \frac{x+5}{-x} = -\frac{x+5}{x} \]

\[ \frac{x^7 + 2x^6 + x^5}{x^3(x+1)^8} = \frac{x^5(x^2 + 2x + 1)}{x^3(x+1)^8} = \frac{\cancel{x^5}(x^2 + 2x + 1)}{\cancel{x^3}(x+1)^8} =\frac{x^2(x^2 + 2x + 1)}{(x+1)^8} =\frac{x^2((x + 1)^2)}{(x+1)^8} =\frac{x^2\cancel{((x + 1)^2)}}{\cancel{(x+1)^8}} =\frac{x^2}{(x+1)^6} \]

\[ \frac{x^2 - 5x - 14}{x^2 - 3x + 2} . \frac{x^2-4}{x^2-14x+49} = \frac{(x + 2)(x - 7)}{(x - 2)(x -1)} . \frac{(x-2)(x+2)}{(x - 7)^2} =\frac{(x + 2)\cancel{(x - 7)}}{\cancel{(x - 2)}(x -1)} . \frac{\cancel{(x-2)}(x+2)}{\cancel{(x - 7)^2}} = \frac{(x + 2)}{(x-1)} . \frac{(x+2)}{(x - 7)} = \frac{(x+2)^2}{(x-1)(x - 7)} \]

\[ \frac{m^2-9}{m^2+5m+6} \div \frac{3-m}{m+2} = \frac{(m+3)(m-3)}{(m + 2)(m + 3)} \div \frac{3-m}{m+2} = \frac{(m+3)(m-3)}{(m + 2)(m + 3)} . \frac{m+2}{3-m} = \frac{(m+3)(m-3)}{(m + 2)(m + 3)} . \frac{m+2}{-(m - 3)} = -\frac{\cancel{(m+3)}\cancel{(m-3)}}{\cancel{(m + 2)}\cancel{(m + 3)}} . \frac{\cancel{(m+2)}}{\cancel{(m - 3)}} = -1 \]

\[ \frac{y^2 + 5y + 4}{\frac{y^2-1}{y+5}} =\frac{(y+1)(y+4)}{\frac{(y-1)(y+1)}{(y+5)}} =\frac{(y+1)(y+4)}{1}.\frac{(y+5)}{(y-1)(y+1)} = \frac{\cancel{(y+1)}(y+4)}{1}.\frac{(y+5)}{(y-1)\cancel{(y+1)}} = \frac{(y+4)(y+5)}{y-1} \]

\[ \frac{4}{6x^2} - \frac{1}{3x^5} + \frac{5}{2x^3} = \frac{4.x^3}{6x^2.x^3} -\frac{1.2}{3x^5.2} + \frac{5.3x^2}{2x^3.3x^2} = \frac{4x^3}{6x^5} - \frac{2}{6x^5} + \frac{15x^2}{6x^5} = \frac{4x^3 + 15x^2 - 2}{6x^5} \]

\[ \frac{2}{z+1}-\frac{z-1}{z+2} =\frac{2.(z+2)}{(z+1).(z+2)}-\frac{(z-1).(z+1)}{(z+2).(z+1)} =\frac{2.(z+2)-((z-1).(z+1))}{(z+2).(z+1)} =\frac{2z+4-z^2+1}{(z+2).(z+1)} =\frac{-z^2+2z+5}{(z+2).(z+1)} \]

\[ \frac{y}{y^2-2y+1} - \frac{2}{y-1} + \frac{3}{y+2} =\frac{y}{(y-1)^2} - \frac{2}{y-1} + \frac{3}{y+2} =\frac{y.(y+2)}{(y-1)^2.(y+2)} - \frac{2.((y-1)(y+2))}{y-1.((y-1)(y+2))} + \frac{3.(y-1)^2}{y+2.(y-1)^2} =\frac{y.(y+2)-2.((y-1)(y+2))+3.(y-1)^2}{(y-1)^2(y+2)} \]
